<?php

上記のコード内の「$artist 」がJSONファイル内の 「id 」と一致すれば、その情報が画面に表示される。 もし「$artist = "all"; 」とすると、今日の日付以降に表示されるすべてのアーティストを表示する。 最初から順番に表示されるので、JSONファイルを日付順に並べることをお勧めする。 イベントを自動的にソートし、日付順に並べます。 不要になったら削除すればよい。

date 「値と 」enddate "値がある。 これらの値によって、複数日のイベントが可能になる。 また、「start 」と 「end 」があり、開始と終了の時間や複数の時間を指定することができます。
1つのイベントに複数のアーティストを追加するには、「 | 」を使用します。 例えば、アップサイクルフェスティバルは以下のようになります： 

"id": "amw|akino|kanasa|bless4"

$today = new DateTime(); /* defaults to "now" */

usort($event, function ($a, $b) {
    return strtotime($a['date']) - strtotime($b['date']);

function dateConvertion($date) {
    $parts = explode('-', $date);

    if (count($parts) !== 3) {
        // Handle invalid date format
        return [
            'month' => '',
            'day'   => '',
            'week'  => ''
        ];
    }

    list($year, $month, $day) = $parts;
    $week = date('w', strtotime($date));
    $day = $day . "日";

    switch ($month) {
        case  1: $month = "1月"; break;
        case  2: $month = "2月"; break;
        case  3: $month = "3月"; break;
        case  4: $month = "4月"; break;
        case  5: $month = "5月"; break;
        case  6: $month = "6月"; break;
        case  7: $month = "7月"; break;
        case  8: $month = "8月"; break;
        case  9: $month = "9月"; break;
        case 10: $month = "10月"; break;
        case 11: $month = "11月"; break;
        case 12: $month = "12月"; break;
    }

    switch ($week) {
        case  0: $week = "(日)"; break;
        case  1: $week = "(月)"; break;
        case  2: $week = "(火)"; break;
        case  3: $week = "(水)"; break;
        case  4: $week = "(木)"; break;
        case  5: $week = "(金)"; break;
        case  6: $week = "(土)"; break;
    }

    return [
        'month' => $month,
        'day'   => $day,
        'week'  => $week,
    ];
}
