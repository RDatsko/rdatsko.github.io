events.js
// Default artist
const artist = "all";

// Escape HTML special characters to prevent XSS
function escapeHTML(str) {
    if (!str) return "";
    return str
        .replace(/&/g, "&")
        .replace(/</g, "<")
        .replace(/>/g, ">")
        .replace(/"/g, """)
        .replace(/'/g, "'");
}

// Convert date into Japanese format
function dateConversion(date) {
    if (!date) {
        return { month: "", day: "", week: "" };
    }
    const parts = date.split("-");
    if (parts.length !== 3) {
        return { month: "", day: "", week: "" };
    }
    let [year, month, day] = parts;
    const d = new Date(date);
    let week = d.getDay();

    day = `${day}日`;

    const months = {
        1: "1月", 2: "2月", 3: "3月", 4: "4月",
        5: "5月", 6: "6月", 7: "7月", 8: "8月",
        9: "9月", 10: "10月", 11: "11月", 12: "12月"
    };

    const weeks = ["(日)", "(月)", "(火)", "(水)", "(木)", "(金)", "(土)"];

    return {
        month: months[parseInt(month, 10)],
        day: day,
        week: weeks[week]
    };
}

// Render events into #events
function renderEvents(events, artist) {
    const container = document.getElementById("events");
    container.innerHTML = ""; // clear first

    // Sort by date
    events.sort((a, b) => new Date(a.date) - new Date(b.date));

    for (let i = 0; i < events.length; i++) {
        const ids = events[i].id.split("|");

        if ((ids.includes(artist) || artist === "all") && (new Date(events[i].date) >= new Date())) {
            const start = dateConversion(events[i].date);
            let { month: smonth, day: sday, week: sweek } = start;

            let end = dateConversion(events[i].enddate);
            let { month: emonth, day: eday, week: eweek } = end;

            const url = escapeHTML(events[i].url);

            let continueStr = "";
            if (!events[i].enddate) {
                emonth = eday = eweek = "";
            } else {
                continueStr = " ~";
                end = dateConversion(events[i].enddate);
                emonth = end.month;
                eday = end.day;
                eweek = end.week;
            }

            container.innerHTML += `
                <div class="col-12 col-md-4 mb_64 article d-flex">
                    <div class="event d-flex flex-column align-self-end justify-content-between">
                        <a href="" class="hover-opacity d-flex" style="pointer-events:none;">
                            <img src="${escapeHTML(events[i].image)}" width="100%" class="hover-opacity">
                        </a>
                        <div class="d-flex flex-column align-self-end" style="width: 100%;">
                            <h6 class="artist">${escapeHTML(events[i].artist)}</h6>
                            <h2 class="title">${escapeHTML(events[i].title)}</h2>
                            <p class="description">${escapeHTML(events[i].description)}</p>
                            <p style="font-size: .75rem; color: #777;">
                                <i class="fa fa-map-marker"></i> ${escapeHTML(events[i].country)}、 ${escapeHTML(events[i].city)}<br>
                                <i class="fa fa-map-location-dot"></i> ${escapeHTML(events[i].place)}<br>
                                <i class="fa fa-calendar"></i> ${smonth} ${sday} ${sweek} ${continueStr}
                                <span style="position: absolute; left: 65%;">
                                    <i class="fa fa-clock"></i> ${escapeHTML(events[i].start)}
                                </span><br>
                                <i></i> ${emonth} ${eday} ${eweek}
                                <span style="position: absolute; left: 65%;">
                                    <i></i> ${escapeHTML(events[i].end)}
                                </span><br>
                            </p>
                            <a href="${url}" target="_blank">
                                詳細はこちら
                                <i class="fa-solid fa-angles-right" aria-hidden="true"></i>
                            </a>
                        </div>
                    </div>
                </div>
            `;
        }
    }
}

// Wait until DOM is ready, then fetch JSON
document.addEventListener("DOMContentLoaded", () => {
    fetch("/kat/assets/data/events.json")
        .then(response => response.json())
        .then(data => {
            // Your PHP did: json_decode(..., true)['events']
            renderEvents(data.events, artist);
        })
        .catch(err => {
            console.error("Failed to load events.json:", err);
        });
});
